    def get_url(method, data)
      base_url = "#{@base_url}/#{method}?"
      params = ""
      data.each {|key, value|
        params += key.to_s + "=" + CGI.escape(value.to_s) + "&" unless key.nil? || value.nil?
      }
      checksum = Digest::SHA1.hexdigest(params.chop + @salt)
      "#{base_url}#{params}checksum=#{checksum}"
    end
